def self.[](key)
  key = key.to_s if key.is_a? Symbol
  setting = if @@cache.has_key?(key) 
      @@cache[key] 
    else
      temp = first(:name => key)
      @@cache[key] = temp.nil? ? nil : temp.value
  end
  setting.nil? ? false : setting
end

def self.[]=(key,value)
  key = key.to_s if key.is_a? Symbol
  @@cache[key] = value
  setting = first_or_create(:name => key)
  setting.value = value
  setting.save
end
