def update
  @user = current_user
  @user.attributes = params[:user]
  
  respond_to do |format|
    if @user.save
      flash[:notice] = 'Password successfully updated'
      format.html { redirect_back_or_default('/') }
    else
      format.html { render(:action => 'edit') }
    end
  end
  
end
