def has_role?(role_name, project = nil)
  @_user_role_names ||= self.roles(:project => project).collect(&:name)
  @_user_role_names.include?(role_name.to_s)
end

def name
  @___name ||= begin
    name = "#{first_name} #{last_name}"
    name = self.login if name.blank?
    name
  end
end
